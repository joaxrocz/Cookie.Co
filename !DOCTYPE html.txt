<script>
    // --- TOTAL CALCULATION ---
    const qtyInputs = document.querySelectorAll('.qty');
    const totalDisplay = document.getElementById('totalDisplay');

    function calculateTotal() {
        let total = 0;
        qtyInputs.forEach(input => {
            const price = parseInt(input.getAttribute('data-price'));
            const qty = parseInt(input.value) || 0;
            total += price * qty;
        });
        totalDisplay.innerText = total;
    }

    qtyInputs.forEach(input => {
        input.addEventListener('input', calculateTotal);
    });

    // --- FORM SUBMISSION ---
    const orderForm = document.getElementById('orderForm');
    
    orderForm.onsubmit = function(e) {
        e.preventDefault();
        
        const total = totalDisplay.innerText;
        if(total == "0") {
            alert("Please select at least one cookie!");
            return;
        }

        const name = document.getElementById('name').value;
        const contact = document.getElementById('contact').value;
        const pickup = document.getElementById('pickupDate').value;

        // Show Summary
        document.getElementById('summaryText').innerHTML = `
            Thank you, <b>${name}</b>!<br>
            Your total is <b>‚Ç±${total}</b>.<br>
            We'll contact you at <b>${contact}</b> for confirmation.
        `;
        
        document.getElementById('summaryModal').style.display = 'flex';
        
        // LOGIC FOR ADMIN: In a real scenario, you'd use EmailJS here.
        console.log("Order submitted for", name, "Total:", total);
    };
